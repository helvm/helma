   57 -- | DeConstructors
   58 safeIOToPTextIO :: Show a => IO (Safe a) -> IO Text
   59 safeIOToPTextIO a = showP <$> safeIOToIO a
   60 
   61 safeIOToIO :: IO (Safe a) -> IO a
   62 safeIOToIO a = safeToIO =<< a
   63 
   64 safeToIO :: Safe a -> IO a
   65 safeToIO = safeTToIO . liftSafe
   66 
   67 safeTToIO :: SafeT IO a -> IO a
   68 safeTToIO = liftExceptT . withExceptT (userError . errorsToString)
   69 
   70 runSafeT :: SafeT m a -> m (Safe a)
   71 runSafeT = runExceptT
   72 
   73 runSafe :: Safe a -> Safe a
   74 runSafe = id
   75 
   76 safeToText :: Safe a -> Text
   77 safeToText (Left messages) = errorsToText messages
   78 safeToText (Right       _) = ""
   79 
   80 safeToEitherLegacy :: Safe a -> EitherLegacy a
   81 safeToEitherLegacy = first errorsToString
   82 
   83 orErrorTuple :: MessageTuple -> Safe a -> a
   84 orErrorTuple t = unsafe . appendErrorTuple t
   85 
   86 orError :: Show e => e -> Safe a -> a
   87 orError e = unsafe . appendError (show e)
   88 
   89 unsafe :: Safe a -> a
   90 unsafe (Right a) = a
   91 unsafe (Left  a) = (error . errorsToText) a
   92 
   93 -- | Constructors
   94 maybeOrError :: Show e => e -> Maybe a -> Safe a
   95 maybeOrError = maybeToSafe . show
   96 
   97 maybeToSafe :: Message -> Maybe a -> Safe a
   98 maybeToSafe = maybeToRight . D.singleton
   99 
  100 safeT :: Monad m => m a -> SafeT m a
  101 safeT a = ExceptT $ pure <$> a
  102 
  103 -- | Lift
  104 liftExceptT :: MonadError e m => ExceptT e m a -> m a
  105 liftExceptT m = liftEither =<< runExceptT m
  106 
  107 liftSafe :: MonadSafe m => Safe a -> m a
  108 liftSafe = liftEither
  109 
  110 liftEitherError :: MonadSafe m => EitherError a -> m a
  111 liftEitherError = liftSafe . first D.singleton
  112 
  113 liftEitherLegacy :: MonadSafe m => EitherLegacy a -> m a
  114 liftEitherLegacy = liftSafe . first stringToErrors

  116 -- | Lift from Maybe
  117 liftMaybeOrErrorTupleList :: MonadSafe m => [MessageTuple] -> Maybe a -> m a
  118 liftMaybeOrErrorTupleList = liftMaybeOrError . tupleListToMessage
  119 
  120 liftMaybeOrErrorTuple :: MonadSafe m => MessageTuple -> Maybe a -> m a
  121 liftMaybeOrErrorTuple = liftMaybeOrError . tupleToMessage

  123 liftMaybeOrError :: MonadSafe m => Message -> Maybe a -> m a
  124 liftMaybeOrError e = liftSafe . maybeToRight (D.singleton e)
  125 
  126 -- | Lift from Message
  127 liftErrorWithTupleList :: MonadSafe m => Message -> [MessageTuple] -> m a
  128 liftErrorWithTupleList m l = liftError (m <> tupleListToMessage l)
  129 
  130 liftErrorTupleList :: MonadSafe m => [MessageTuple] -> m a
  131 liftErrorTupleList = liftError . tupleListToMessage
  132 
  133 liftErrorWithPrefix :: MonadSafe m => Message -> Message -> m a
  134 liftErrorWithPrefix prefix showed = liftErrorTuple (prefix , showed)
  135 
  136 liftErrorTuple :: MonadSafe m => MessageTuple -> m a
  137 liftErrorTuple = liftError . tupleToMessage
  138 
  139 liftError :: MonadSafe m => Message -> m a
  140 liftError = throwError . D.singleton
  141 
  142 -- | Append Message
  143 appendErrorTupleList :: MonadSafe m => [MessageTuple] -> m a -> m a
  144 appendErrorTupleList = appendError . tupleListToMessage
  145 
  146 appendErrorTuple :: MonadSafe m => MessageTuple -> m a -> m a
  147 appendErrorTuple = appendError . tupleToMessage
  148 
  149 appendError :: MonadSafe m => Message -> m a -> m a
  150 appendError message a = catchError a appendAndThrow where appendAndThrow es = throwError (es `D.snoc` message)
