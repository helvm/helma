   35 -- | Internal function
   36 listFromDescList :: Default a => IndexedList a -> [a]
   37 listFromDescList = flip go [] where
   38   go :: Default a => IndexedList a -> [a] -> [a]
   39   go []                          acc = acc
   40   go [(i , v)]                   acc = consDef i $ v : acc
   41   go ((i1 , v1) : (i2 , v2) : l) acc = go ((i2 , v2) : l) $ consDef (i1 - i2 - 1)  $ v1 : acc
   42 
   43 consDef :: Default a => Key -> [a] -> [a]
   44 consDef i l = (check . compare i) 0 where
   45   check LT = error "MapList.consDef index is negative"
   46   check EQ = l
   47   check GT = consDef (i - 1) (def : l)
