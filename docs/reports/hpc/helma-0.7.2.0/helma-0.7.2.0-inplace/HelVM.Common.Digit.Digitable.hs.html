   29 toBitsBySize :: Int -> Int -> [Natural]
   30 toBitsBySize 0    _ = []
   31 toBitsBySize size 0 = [0 | _ <- [1..size]]
   32 toBitsBySize size x
   33  | k == 0    = 0 : toBitsBySize maxDigit x
   34  | otherwise = 1 : toBitsBySize maxDigit (x - k*m)
   35     where
   36       maxDigit = size - 1
   37       m = 2 ^ maxDigit
   38       k = x `div` m
