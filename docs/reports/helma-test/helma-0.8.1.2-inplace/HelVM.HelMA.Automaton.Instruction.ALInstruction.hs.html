   43 -- | Types
   44 data ALInstruction =
   45     Cons      Integer
   46   | Unary    !UnaryInstruction
   47   | Binary   !BinaryInstruction
   48   | Binaries [BinaryInstruction]
   49   | SStatic  !StackIndex !ManipulationInstruction
   50   | SDynamic             !ManipulationInstruction
   51   | SIO      !IOInstruction
   52   | Halibut
   53   | Pick
   54   | Discard
   55   deriving stock (Eq , Read , Show)

   57 dupI , swapI , rotI :: ALInstruction
   58 dupI  = SStatic 0 Copy
   59 swapI = SStatic 1 Move
