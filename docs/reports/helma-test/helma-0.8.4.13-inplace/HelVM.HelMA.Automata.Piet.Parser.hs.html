   53 imageGuessCodelLengthDefault :: IO Int
   54 imageGuessCodelLengthDefault = do
   55   width  <- imageGetWidth
   56   height <- imageGetHeight
   57   rows   <- mapM (\y -> mapM (`imageQueryPixel` y) [0 .. width -1]) [0 .. height-1]
   58   cols   <- mapM (\x -> mapM ( imageQueryPixel  x) [0 .. height-1]) [0 .. width -1]
   59   pure $ imageGuessCodelLengthDefault' rows cols width height

   61 imageGuessCodelLengthDefault' :: (Eq a1, Eq a2) => [a1] -> [a2] -> Int -> Int -> Int
   62 imageGuessCodelLengthDefault' rows cols width height = findJust (== 1) $ scanl gcd (gcd width height) list
   63   where list = groupAndLength rows <> groupAndLength cols

   71 imageFromContextWithMax :: Int -> IO ColorImage
   72 imageFromContextWithMax codelLength = do

   74   height <- (`div` codelLength) <$> imageGetHeight
   75   pixels <- mapM (toColors codelLength) [ (x, y) | x <- [ 0 .. width-1 ], y <- [ 0 .. height-1 ] ]
