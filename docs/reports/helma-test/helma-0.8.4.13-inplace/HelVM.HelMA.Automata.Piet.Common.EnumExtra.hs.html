    8 rotateEnum :: (Bounded a , Enum a) => Int -> a -> a
    9 rotateEnum n x = unsafeEnum $ (fromEnum x + n) `mod` (fromEnum maxBound_ - fromEnum minBound_ + 1)
   10   where
   11     maxBound_ = asTypeOf maxBound x
   12     minBound_ = asTypeOf minBound x
