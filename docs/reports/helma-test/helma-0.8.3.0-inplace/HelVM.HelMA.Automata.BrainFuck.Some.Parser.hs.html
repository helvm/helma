   45 moveRParser :: ReadP SomeInstruction

   48 moveLParser :: ReadP SomeInstruction

   51 incParser :: ReadP SomeInstruction
   52 incParser = Inc 1 <$ char '+'

   54 decParser :: ReadP SomeInstruction

   57 outParser :: ReadP SomeInstruction

   60 inParser :: ReadP SomeInstruction

   63 whileParser :: ReadP SomeInstruction

   66 filterComments :: Source -> Source
   67 filterComments = Text.filter isNotComment

   69 isNotComment :: Char -> Bool
