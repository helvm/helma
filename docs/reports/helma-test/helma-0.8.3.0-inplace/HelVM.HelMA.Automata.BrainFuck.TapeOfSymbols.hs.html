   35 changeTape :: FullTapeD e -> FullTapeD e -> Int -> FullTapeD e
   36 changeTape lf gf i t = loop atc (i , t) where
   37   atc (i' , t') = (check . compare0) i' where
   38     check LT = Left (i' - 1 , lf t')
   39     check GT = Left (i' + 1 , gf t')
   40     check EQ = Right t'
