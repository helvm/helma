   21 evalSource :: (BIO m , Symbol e) => Source -> FullTape e -> LoopLimit -> DumpType -> m ()
   22 evalSource source tape _limit dt = logDump dt =<< doInstruction (Automaton ([] , tokenize source) tape)

   24 doInstruction :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   25 doInstruction a = checkOpt $ currentInstruction a where
   26   checkOpt Nothing  = doEnd         a
   27   checkOpt (Just i) = check i
   28   check (Simple MoveR ) = doInstruction $ moveRAutomaton a
   29   check (Simple MoveL ) = doInstruction $ moveLAutomaton a
   30   check (Simple Inc   ) = doInstruction $ incAutomaton a
   31   check (Simple Dec   ) = doInstruction $ decAutomaton a
   32   check (Simple Output) = doOutputChar  a
   33   check (Simple Input ) = doInstruction =<< doInputChar a
   34   check  JmpPast        = doInstruction (doJmpPast     a)
   35   check  JmpBack        = doInstruction (doJmpBack     a)
   36 
   37 -- | Control instruction
   38 doJmpPast :: Symbol e => Automaton e -> Automaton e
   39 doJmpPast (Automaton table tape@(_ , 0 : _)) = Automaton (Table.jumpPast table) tape
   40 doJmpPast (Automaton table tape            ) = Automaton (Table.nextInst table) tape
   41 
   42 doJmpBack :: Symbol e => Automaton e -> Automaton e
   43 doJmpBack (Automaton table tape@(_ , 0 : _)) = Automaton (Table.nextInst table) tape
   44 doJmpBack (Automaton table tape            ) = Automaton (Table.jumpBack table) tape

   46 -- | IO instructions
   47 doOutputChar :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   48 doOutputChar   (Automaton _ (_ ,    [])) = error "Illegal State"
   49 doOutputChar a@(Automaton _ (_ , e : _)) = wPutChar (toChar e) *> doInstruction (nextInstAutomaton a)

   51 doInputChar :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   52 doInputChar a = (newAutomatonForChar a) <$> wGetChar

   82 updateTable :: (Table -> Table) -> Automaton e -> Automaton e
