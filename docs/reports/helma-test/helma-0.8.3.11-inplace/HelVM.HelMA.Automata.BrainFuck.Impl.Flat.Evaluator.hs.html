   28 doInstruction :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   29 doInstruction a = checkOpt $ currentInstruction a where
   30   checkOpt Nothing  = doEnd         a
   31   checkOpt (Just i) = check i
   32   check (Simple MoveR ) = doInstruction $ moveRAutomaton a
   33   check (Simple MoveL ) = doInstruction $ moveLAutomaton a
   34   check (Simple Inc   ) = doInstruction $ incAutomaton   a
   35   check (Simple Dec   ) = doInstruction $ decAutomaton   a
   36   check (Simple Output) = doOutputChar  a
   37   check (Simple Input ) = doInstruction =<< doInputChar  a
   38   check  JmpPast        = doInstruction =<< (doJmpPast   a)
   39   check  JmpBack        = doInstruction =<< (doJmpBack   a)

   41 -- | Control instruction
   42 doJmpPast :: (MonadSafe m , Symbol e) => Automaton e -> m $ Automaton e
   43 doJmpPast = teeMap (flip doJmpPast') currentSymbolSafe

   45 doJmpPast' :: (Eq a, Num a) => a -> Automaton e -> Automaton e
   46 doJmpPast' 0 = updateTable Table.jumpPast

   49 doJmpBack :: (MonadSafe m , Symbol e) => Automaton e -> m $ Automaton e

   52 doJmpBack' :: (Eq a, Num a) => a -> Automaton e -> Automaton e
   53 doJmpBack' 0 = nextInstAutomaton

   92 moveRAutomaton :: Symbol e => Automaton e -> Automaton e
   93 moveRAutomaton = nextInstAutomaton . updateTape Tape.moveHeadRight

   95 moveLAutomaton :: Symbol e => Automaton e -> Automaton e
   96 moveLAutomaton = nextInstAutomaton . updateTape Tape.moveHeadLeft

   98 incAutomaton :: Symbol e => Automaton e -> Automaton e
   99 incAutomaton = nextInstAutomaton . updateTape Tape.nextSymbol

  101 decAutomaton :: Symbol e => Automaton e -> Automaton e

  107 updateTable :: (Table -> Table) -> Automaton e -> Automaton e
  108 updateTable f a = a { unitTable = f $ unitTable a }

  110 updateTape :: (FullTape e1 -> FullTape e2) -> Automaton e1 -> Automaton e2
  111 updateTape f a = a { unitTape = f $ unitTape a }

  119 teeMap :: Functor f => (t -> a -> b) -> (t -> f a) -> t -> f b
  120 teeMap f2 f1 x = f2 x <$> f1 x
