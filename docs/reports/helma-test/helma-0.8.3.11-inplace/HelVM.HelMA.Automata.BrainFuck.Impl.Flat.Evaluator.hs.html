   27 evalSource :: (BIO m , Symbol e) => Source -> FullTape e -> LoopLimit -> DumpType -> m ()
   28 evalSource source tape limit dt = logDump dt =<< run limit (Automaton ([] , tokenize source) tape)

   30 run :: (BIO m , Symbol e) => LoopLimit -> Automaton e -> m $ Automaton e
   31 run = loopMWithLimit nextState
   32 
   33 nextState :: (BIO m , Symbol e) => Automaton e -> m $ AutomatonSame e
   34 nextState = tee (flip doInstructionOpt) currentInstruction
   35 
   36 doInstructionOpt :: (BIO m , Symbol e) => Maybe FlatInstruction -> Automaton e -> m $ AutomatonSame e
   37 doInstructionOpt (Just i) = Right <.> doInstruction i
   38 doInstructionOpt Nothing  = Left <.> pure
   39 
   40 doInstruction :: (BIO m , Symbol e) => FlatInstruction -> Automaton e -> m $ Automaton e
   41 doInstruction  JmpPast           = doJmpPast
   42 doInstruction  JmpBack           = doJmpBack
   43 doInstruction (Simple Output)    = doOutputChar
   44 doInstruction (Simple Input )    = doInputChar
   45 doInstruction (Simple (Pure i )) = pure . nextInstAutomaton . updateTape (doPure i)
   46 
   47 -- | Control instruction
   48 doJmpPast :: (MonadSafe m , Symbol e) => Automaton e -> m $ Automaton e
   49 doJmpPast = teeMap (flip doJmpPastForValue) currentSymbolSafe
   50 
   51 doJmpPastForValue :: (Eq a, Num a) => a -> Automaton e -> Automaton e
   52 doJmpPastForValue 0 = updateTable Table.jumpPast
   53 doJmpPastForValue _ = nextInstAutomaton
   54 
   55 doJmpBack :: (MonadSafe m , Symbol e) => Automaton e -> m $ Automaton e
   56 doJmpBack = teeMap (flip doJmpBackForValue) currentSymbolSafe
   57 
   58 doJmpBackForValue :: (Eq a, Num a) => a -> Automaton e -> Automaton e
   59 doJmpBackForValue 0 = nextInstAutomaton
   60 doJmpBackForValue _ = updateTable Table.jumpBack

   72 -- | Pure Instruction
   73 doPure :: Symbol e => PureInstruction -> FullTapeD e
   74 doPure MoveR = Tape.moveHeadRight
   75 doPure MoveL = Tape.moveHeadLeft
   76 doPure Inc   = Tape.nextSymbol
   77 doPure Dec   = Tape.prevSymbol

   95 updateTable :: (Table -> Table) -> Automaton e -> Automaton e
   96 updateTable f a = a { unitTable = f $ unitTable a }

   98 updateTape :: (FullTape e -> FullTape e) -> Automaton e -> Automaton e
   99 updateTape f a = a { unitTape = f $ unitTape a }
