   24 evalSource :: (BIO m , Symbol e) => Source -> FullTape e -> LoopLimit -> DumpType -> m ()
   25 evalSource source tape _limit dt = logDump dt =<< doInstruction (Automaton ([] , tokenize source) tape)

   27 doInstruction :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   28 doInstruction a = checkOpt $ currentInstruction a where
   29   checkOpt Nothing  = doEnd         a
   30   checkOpt (Just i) = check i
   31   check (Simple MoveR ) = doInstruction $ moveRAutomaton a
   32   check (Simple MoveL ) = doInstruction $ moveLAutomaton a
   33   check (Simple Inc   ) = doInstruction $ incAutomaton a
   34   check (Simple Dec   ) = doInstruction $ decAutomaton a
   35   check (Simple Output) = doOutputChar  a
   36   check (Simple Input ) = doInstruction =<< doInputChar a
   37   check  JmpPast        = doInstruction (doJmpPast     a)
   38   check  JmpBack        = doInstruction (doJmpBack     a)
   39 
   40 -- | Control instruction
   41 doJmpPast :: Symbol e => Automaton e -> Automaton e
   42 doJmpPast = tee (flip doJmpPast') currentSymbol

   44 doJmpPast' :: (Eq a, Num a) => a -> Automaton e -> Automaton e
   45 doJmpPast' 0 = updateTable Table.jumpPast
   46 doJmpPast' _ = nextInstAutomaton

   48 doJmpBack :: Symbol e => Automaton e -> Automaton e
   49 doJmpBack = tee (flip doJmpBack') currentSymbol

   51 doJmpBack' :: (Eq a, Num a) => a -> Automaton e -> Automaton e
   52 doJmpBack' 0 = nextInstAutomaton
   53 doJmpBack' _ = updateTable Table.jumpBack

   55 -- | IO instructions
   56 doOutputChar :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   57 doOutputChar   (Automaton _ (_ ,    [])) = error "Illegal State"
   58 doOutputChar a@(Automaton _ (_ , e : _)) = wPutChar (toChar e) *> doInstruction (nextInstAutomaton a)

   60 doInputChar :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   61 doInputChar a = newAutomatonForChar a <$> wGetChar

   63 -- | Terminate instruction

   70 currentSymbol :: Automaton e -> e

   73 currentInstruction :: Automaton e -> Maybe FlatInstruction

   76 newAutomatonForChar :: Symbol e => Automaton e -> Char -> Automaton e
   77 newAutomatonForChar (Automaton table tape) = Automaton (Table.nextInst table) . flip Tape.writeSymbol tape

   79 moveRAutomaton :: Symbol e => Automaton e -> Automaton e
   80 moveRAutomaton = nextInstAutomaton . updateTape Tape.moveHeadRight
   81 
   82 moveLAutomaton :: Symbol e => Automaton e -> Automaton e
   83 moveLAutomaton = nextInstAutomaton . updateTape Tape.moveHeadLeft
   84 
   85 incAutomaton :: Symbol e => Automaton e -> Automaton e
   86 incAutomaton = nextInstAutomaton . updateTape Tape.nextSymbol
   87 
   88 decAutomaton :: Symbol e => Automaton e -> Automaton e
   89 decAutomaton = nextInstAutomaton . updateTape Tape.prevSymbol

   91 nextInstAutomaton :: Automaton e -> Automaton e
   92 nextInstAutomaton = updateTable Table.nextInst

   94 updateTable :: (Table -> Table) -> Automaton e -> Automaton e
   95 updateTable f a = a { unitTable = f $ unitTable a }

   97 updateTape :: (FullTape e1 -> FullTape e2) -> Automaton e1 -> Automaton e2
   98 updateTape f a = a { unitTape = f $ unitTape a }
