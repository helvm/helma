   27 evalSource :: (BIO m , Symbol e) => Source -> FullTape e -> LoopLimit -> DumpType -> m ()
   28 evalSource source tape _limit dt = logDump dt =<< run (Automaton ([] , tokenize source) tape)

   30 run :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   31 run = loopM nextState

   33 nextState :: (BIO m , Symbol e) => Automaton e -> m $ AutomatonSame e
   34 nextState = tee (flip doInstructionOpt) currentInstruction

   36 doInstructionOpt :: (BIO m , Symbol e) => Maybe FlatInstruction -> Automaton e -> m $ AutomatonSame e
   37 doInstructionOpt (Just i) = Right <.> doInstruction i
   38 doInstructionOpt Nothing  = Left <.> doEnd

   40 doInstruction :: (BIO m , Symbol e) => FlatInstruction -> Automaton e -> m $ Automaton e
   41 doInstruction (Simple MoveR ) = pure . moveRAutomaton
   42 doInstruction (Simple MoveL ) = pure . moveLAutomaton
   43 doInstruction (Simple Inc   ) = pure . incAutomaton
   44 doInstruction (Simple Dec   ) = pure . decAutomaton
   45 doInstruction (Simple Output) = doOutputChar
   46 doInstruction (Simple Input ) = doInputChar
   47 doInstruction  JmpPast        = doJmpPast
   48 doInstruction  JmpBack        = doJmpBack

   50 -- | Control instruction
   51 doJmpPast :: (MonadSafe m , Symbol e) => Automaton e -> m $ Automaton e
   52 doJmpPast = teeMap (flip doJmpPast') currentSymbolSafe
   53 
   54 doJmpPast' :: (Eq a, Num a) => a -> Automaton e -> Automaton e
   55 doJmpPast' 0 = updateTable Table.jumpPast
   56 doJmpPast' _ = nextInstAutomaton
   57 
   58 doJmpBack :: (MonadSafe m , Symbol e) => Automaton e -> m $ Automaton e
   59 doJmpBack = teeMap (flip doJmpBack') currentSymbolSafe
   60 
   61 doJmpBack' :: (Eq a, Num a) => a -> Automaton e -> Automaton e
   62 doJmpBack' 0 = nextInstAutomaton
   63 doJmpBack' _ = updateTable Table.jumpBack

   65 -- | IO instructions

   67 doOutputChar a = (nextInstAutomaton a) <$ wPutSymbol a

   69 wPutSymbol :: (BIO m , Symbol e) => Automaton e -> m ()

   91 moveRAutomaton :: Symbol e => Automaton e -> Automaton e
   92 moveRAutomaton = nextInstAutomaton . updateTape Tape.moveHeadRight

   94 moveLAutomaton :: Symbol e => Automaton e -> Automaton e
   95 moveLAutomaton = nextInstAutomaton . updateTape Tape.moveHeadLeft

   97 incAutomaton :: Symbol e => Automaton e -> Automaton e
   98 incAutomaton = nextInstAutomaton . updateTape Tape.nextSymbol

  100 decAutomaton :: Symbol e => Automaton e -> Automaton e
  101 decAutomaton = nextInstAutomaton . updateTape Tape.prevSymbol

  103 nextInstAutomaton :: Automaton e -> Automaton e

  106 updateTable :: (Table -> Table) -> Automaton e -> Automaton e
  107 updateTable f a = a { unitTable = f $ unitTable a }

  109 updateTape :: (FullTape e1 -> FullTape e2) -> Automaton e1 -> Automaton e2
  110 updateTape f a = a { unitTape = f $ unitTape a }

  118 teeMap :: Functor f => (t -> a -> b) -> (t -> f a) -> t -> f b
  119 teeMap f2 f1 x = f2 x <$> f1 x
