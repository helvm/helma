   21 evalSource :: (BIO m , Symbol e) => Source -> FullTape e -> LoopLimit -> DumpType -> m ()
   22 evalSource source tape _limit dt = logDump dt =<< flip runList tape =<< parseWithOptimize source

   24 runList :: (BIO m , Symbol e) => FastInstructionList -> FullTape e -> m $ Automaton e
   25 runList il = nextStep (IU il 0)
   26 
   27 nextStep :: (BIO m , Symbol e) => InstructionUnit -> FullTape e -> m $ Automaton e
   28 nextStep (IU iv ic) = doInstructionOpt (iv `indexMaybe` ic) (IU iv $ ic + 1)
   29 
   30 doInstructionOpt :: (BIO m , Symbol e) => Maybe FastInstruction -> InstructionUnit -> FullTape e -> m $ Automaton e
   31 doInstructionOpt (Just i) table tape = doInstruction i table tape
   32 doInstructionOpt Nothing  table tape = doEnd           (Automaton table tape )
   33 
   34 doInstruction :: (BIO m , Symbol e) => FastInstruction -> InstructionUnit -> FullTape e -> m $ Automaton e
   35 doInstruction (Move   i       ) table tape       = nextStep     table (moveHead          i        tape)
   36 doInstruction (Inc    i       ) table tape       = nextStep     table (incSymbol         i        tape)
   37 doInstruction  Output           table tape       = doOutputChar table                             tape
   38 doInstruction  Input            table tape       = doInputChar  table                             tape
   39 doInstruction (While  iv      ) table tape       = doWhile iv   table                             tape
   40 doInstruction (Set    i       ) table tape       = nextStep     table (setSymbol         i        tape)
   41 
   42 doInstruction (SubClr          f    ) table tape = nextStep table (subAndClearSymbol          f     tape)
   43 doInstruction (AddClr          f    ) table tape = nextStep table (addAndClearSymbol          f     tape)
   44 doInstruction (MulAddClr m     f    ) table tape = nextStep table (mulAddAndClearSymbol m     f     tape)
   45 
   46 doInstruction (DupClr          f1 f2) table tape = nextStep table (dupAndClearSymbol          f1 f2 tape)
   47 doInstruction (MulDupClr m1 m2 f1 f2) table tape = nextStep table (mulDupAndClearSymbol m1 m2 f1 f2 tape)
   48 
   49 doInstruction (TriClr i1 i2 i3) table tape       = nextStep     table (triAndClearSymbol i1 i2 i3 tape)
   50 
   51 doWhile :: (BIO m , Symbol e) => FastInstructionList -> InstructionUnit -> FullTape e -> m $ Automaton e
   52 doWhile _  table tape@(_ , 0:_) = nextStep table tape
   53 doWhile iv table tape           = doWhileWithTape =<< runList iv tape where
   54   doWhileWithTape :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   55   doWhileWithTape = doWhile iv table . unitTape

   57 -- | IO instructions
   58 doOutputChar :: (BIO m , Symbol e) => InstructionUnit -> FullTape e -> m $ Automaton e
   59 doOutputChar _          (_ ,  []) = error "Illegal State"
   60 doOutputChar table tape@(_ , e:_) = wPutChar (toChar e) *> nextStep table tape
   61                                                                              
   62 doInputChar  :: (BIO m , Symbol e) => InstructionUnit -> FullTape e -> m $ Automaton e
   63 doInputChar table tape = (nextStep table . flip writeSymbol tape) =<< wGetChar
