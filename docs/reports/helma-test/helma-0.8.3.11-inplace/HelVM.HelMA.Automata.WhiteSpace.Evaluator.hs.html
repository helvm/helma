   45 evalParams :: BIO m => TokenType -> EvalParams -> m ()
   46 evalParams tokenType p = eval tokenType (source p) (formatType p) (stackAutoOptions p) (ramAutoOptions p) Nothing (dumpAutoOptions p)

   48 eval :: BIO m => TokenType -> Source -> FormatType -> StackType -> RAMType -> LoopLimit -> DumpType -> m ()
   49 eval tokenType source = evalTL $ tokenize tokenType source
   50 
   51 evalTL :: BIO m => TokenList -> FormatType -> StackType -> RAMType -> LoopLimit -> DumpType -> m ()
   52 evalTL tl ascii st rt limit dt = evalTL' =<< liftSafe (parseFromTL ascii tl) where evalTL' il = evalIL il st rt limit dt
   53 
   54 evalIL :: BIO m => InstructionList -> StackType -> RAMType -> LoopLimit -> DumpType -> m ()
   55 evalIL il s ListRAMType    = evalIL' il s []
   56 evalIL il s SeqRAMType     = evalIL' il s Seq.empty
   57 evalIL il s SListRAMType   = evalIL' il s SList.sListEmpty
   58 evalIL il s MapListRAMType = evalIL' il s MapList.mapListEmpty
   59 
   60 evalIL' :: (RAutomatonIO Symbol r m) => InstructionList -> StackType -> r -> LoopLimit -> DumpType -> m ()
   61 evalIL' il ListStackType  = start il []
   62 evalIL' il SeqStackType   = start il Seq.empty
   63 evalIL' il SListStackType = start il SList.sListEmpty
