   23 evalSource :: (BIO m , Symbol e) => Source -> FullTape e -> LoopLimit -> DumpType -> m ()
   24 evalSource source tape _limit dt = logDump dt =<< flip runVector tape =<< parseAsVector source

   26 runVector :: (BIO m , Symbol e) => TreeInstructionVector -> FullTape e -> m $ Automaton e
   27 runVector iv = nextStep (IU iv 0)
   28                                                                                                   
   29 nextStep :: (BIO m , Symbol e) => InstructionUnit -> FullTape e -> m $ Automaton e
   30 nextStep  (IU iv ic) tape = doInstructionOpt (iv `indexMaybe` ic) (Automaton (IU iv $ ic + 1) tape)
   31 
   32 doInstructionOpt :: (BIO m , Symbol e) => Maybe TreeInstruction -> Automaton e -> m $ Automaton e
   33 doInstructionOpt (Just i) = doInstruction i
   34 doInstructionOpt  Nothing = pure
   35 
   36 doInstruction :: (BIO m , Symbol e) => TreeInstruction -> Automaton e -> m $ Automaton e
   37 doInstruction (Simple MoveR ) (Automaton table tape) = nextStep      table (moveHeadRight tape)
   38 doInstruction (Simple MoveL ) (Automaton table tape) = nextStep      table  (moveHeadLeft tape)
   39 doInstruction (Simple Inc   ) (Automaton table tape) = nextStep      table    (nextSymbol tape)
   40 doInstruction (Simple Dec   ) (Automaton table tape) = nextStep      table    (prevSymbol tape)
   41 doInstruction (Simple Output) a                      = doOutputChar a
   42 doInstruction (Simple Input ) a                      = doInputChar  a
   43 doInstruction (While  iv    ) a                      = doWhile iv   a
   44 
   45 -- | Control Instruction
   46 doWhile :: (BIO m , Symbol e) => TreeInstructionVector -> Automaton e -> m $ Automaton e
   47 doWhile _  (Automaton table tape@(_ , 0:_)) = nextStep table tape
   48 doWhile iv (Automaton table tape          ) = doWhileWithTape =<< runVector iv tape where
   49   doWhileWithTape :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   50   doWhileWithTape (Automaton _ tape') = doWhile iv (Automaton table tape')
   51 
   52 -- | IO instructions                                                                     
   53 doInputChar  :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   54 doInputChar (Automaton table tape) = (nextStep table . flip writeSymbol tape) =<< wGetChar
   55 
   56 doOutputChar :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e                     
   57 doOutputChar (Automaton _          (_ ,  [])) = error "Illegal State"
   58 doOutputChar (Automaton table tape@(_ , e:_)) = wPutChar (toChar e) *> nextStep table tape
