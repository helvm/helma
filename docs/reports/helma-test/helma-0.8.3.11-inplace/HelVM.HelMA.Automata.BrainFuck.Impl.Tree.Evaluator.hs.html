   27 evalSource :: (BIO m , Symbol e) => Source -> FullTape e -> LoopLimit -> DumpType -> m ()
   28 evalSource source tape _limit dt = logDump dt =<< flip runVector tape =<< parseAsVector source

   30 runVector :: (BIO m , Symbol e) => TreeInstructionVector -> FullTape e -> m $ Automaton e
   31 runVector iv tape = nextStep $ newAutomaton iv tape
   32 
   33 nextStepDeprecated :: (BIO m , Symbol e) => InstructionUnit -> FullTape e -> m $ Automaton e
   34 nextStepDeprecated iu tape = nextStep (Automaton iu tape)

   36 nextStep :: (BIO m , Symbol e) => Automaton e -> m $ Automaton e
   37 nextStep a = doInstructionOpt (currentInstruction a) (nextIC a)
   38 
   39 doInstructionOpt :: (BIO m , Symbol e) => Maybe TreeInstruction -> Automaton e -> m $ Automaton e
   40 doInstructionOpt (Just i) = doInstruction i
   41 doInstructionOpt  Nothing = pure
   42 
   43 doInstruction :: (BIO m , Symbol e) => TreeInstruction -> Automaton e -> m $ Automaton e
   44 doInstruction (While  iv    )    = doWhile iv
   45 doInstruction (Simple Output)    = doOutputChar
   46 doInstruction (Simple Input )    = doInputChar
   47 doInstruction (Simple (Pure i )) = nextStep . updateTape (doPure i)

   49 -- | Control Instruction
   50 doWhile :: (BIO m , Symbol e) => TreeInstructionVector -> Automaton e -> m $ Automaton e
   51 doWhile _  a@(Automaton _ (_ , 0 : _)) = nextStep a
   52 doWhile iv a                           = doWhileWithTape iv (unitUI a) =<< runVector iv (unitTape a)

   59 doWhileWithTape iv table (Automaton _ tape) = doWhile iv (Automaton table tape)

   99 updateUI :: (InstructionUnit -> InstructionUnit) -> Automaton e -> Automaton e
  100 updateUI f a = a { unitUI = f $ unitUI a }
