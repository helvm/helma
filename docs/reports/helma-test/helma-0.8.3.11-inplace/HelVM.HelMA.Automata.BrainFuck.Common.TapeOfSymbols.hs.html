   36 triAndClearSymbol :: (Symbol e) => Integer -> Integer -> Integer -> FullTapeD e
   37 triAndClearSymbol f1 f2 f3 tape = tape & stepSymbol f1 & stepSymbol f2 & stepSymbol f3 & backAndClear back where
   38   back = negate (f1 + f2 + f3)
   39   stepSymbol = step symbol
   40   symbol = readSymbol tape
   41 
   42 mulDupAndClearSymbol :: (Symbol e) => Integer -> Integer -> Integer -> Integer -> FullTapeD e
   43 mulDupAndClearSymbol m1 m2 f1 f2 tape = tape & step ms1 f1 & step ms2 f2 & backAndClear back where
   44   back = negate (f1 + f2)
   45   ms1 = symbol * fromIntegral m1
   46   ms2 = symbol * fromIntegral m2
   47   symbol = readSymbol tape
   48 
   49 dupAndClearSymbol :: (Symbol e) => Integer -> Integer -> FullTapeD e
   50 dupAndClearSymbol f1 f2 tape = tape & stepSymbol f1 & stepSymbol f2 & backAndClear back where
   51   back = negate (f1 + f2)
   52   stepSymbol = step symbol
   53   symbol = readSymbol tape

   55 mulAddAndClearSymbol :: (Symbol e) => Integer -> Integer -> FullTapeD e
   56 mulAddAndClearSymbol mul forward tape = tape & step mulSymbol forward & backAndClear back where
   57   back = negate forward
   58   mulSymbol = symbol * fromIntegral mul
   59   symbol = readSymbol tape

   61 addAndClearSymbol :: (Symbol e) => Integer -> FullTapeD e
   62 addAndClearSymbol = changeAndClearSymbol id

   64 subAndClearSymbol :: (Symbol e) => Integer -> FullTapeD e
   65 subAndClearSymbol = changeAndClearSymbol negate

   67 changeAndClearSymbol :: (Symbol e) => (e -> e) -> Integer -> FullTapeD e
   68 changeAndClearSymbol f forward tape = tape & step symbol forward & backAndClear back where
   69   back = negate forward
   70   symbol = f $ readSymbol tape

   72 step :: (Symbol e) => e -> Integer -> FullTapeD e
   73 step symbol forward = addSymbol symbol . moveHead forward

   75 backAndClear :: (Symbol e) => Integer -> FullTapeD e
   76 backAndClear back = clearSymbol . moveHead back

  101 modifyCell :: D e -> FullTapeD e
  102 modifyCell f (left , cell : right) = (left , f cell : right)
  103 modifyCell _ (_ , [])              = error "End of the Tape"
  104 
  105 readSymbol :: FullTape e -> e
  106 readSymbol (_ , cell : _) = cell
  107 readSymbol (_ , [])       = error "End of the Tape"

  109 -- | Moves

  111 moveHead :: (Symbol e) => Integer -> FullTapeD e
  112 moveHead = changeTape moveHeadRight moveHeadLeft
  113 
  114 changeTape :: FullTapeD e -> FullTapeD e -> Integer -> FullTapeD e
  115 changeTape lf gf i t = loop atc (i , t) where
  116   atc (i' , t') = (check . compare0) i' where
  117     check LT = Left (i' - 1 , lf t')
  118     check GT = Left (i' + 1 , gf t')
  119     check EQ = Right t'
  120 
  121 moveHeadRight :: (Symbol e) => FullTapeD e
  122 moveHeadRight (cell : left , right) = pad (left , cell : right)
  123 moveHeadRight ([] , _)              = error "End of the Tape"
  124 
  125 moveHeadLeft :: (Symbol e) => FullTapeD e
  126 moveHeadLeft (left , cell : right) = pad (cell : left , right)
  127 moveHeadLeft (_ , [])              = error "End of the Tape"
  128 
  129 pad :: (Symbol e) => FullTapeD e
  130 pad ([] , [])    = newTape
  131 pad ([] , right) = ([def] , right)
  132 pad (left , [])  = (left , [def])
  133 pad tape         = tape
