   33 evalSource :: BIO m => Source -> RAMType -> LoopLimit -> DumpType -> m ()
   34 evalSource source = evalIL $ tokenize source
   35 
   36 evalIL :: AutomatonIO e m => [e] -> RAMType -> LoopLimit -> DumpType -> m ()
   37 evalIL = flip evalIL'
   38 
   39 evalIL' :: AutomatonIO e m => RAMType -> [e] -> LoopLimit -> DumpType -> m ()
   40 evalIL' ListRAMType    = start
   41 evalIL' SeqRAMType     = start . Seq.fromList
   42 evalIL' SListRAMType   = start . SList.sListFromList
   43 evalIL' MapListRAMType = start . MapList.mapListFromList
   44 
   45 start :: RAutomatonIO e r m => r -> LoopLimit -> DumpType -> m ()
   46 start r limit dt = logDump dt =<< run limit (newAutomaton r)
