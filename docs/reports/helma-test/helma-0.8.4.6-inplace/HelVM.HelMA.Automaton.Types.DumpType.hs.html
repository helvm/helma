   10 logDump :: (BIO m , Show d) => DumpType -> d -> m ()
   11 logDump dt d = logDump' $ dump dt d where
   12   logDump' Nothing  = pass
   13   logDump' (Just t) = logMessageTuple ("dump" , t)
