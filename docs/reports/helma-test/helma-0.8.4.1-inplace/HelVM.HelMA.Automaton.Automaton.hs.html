   27 run :: (SRAutomatonIO Symbol s r m) => Maybe Natural -> Automaton s r -> m $ Automaton s r
   28 run = loopMWithLimit nextState
   29 
   30 ----
   31 
   32 nextState :: (SRAutomatonIO Symbol s r m) => Automaton s r -> m $ AutomatonSame s r
   33 nextState u@(Automaton cu _ _) = nextStateForInstruction =<< currentInstruction cu where
   34   nextStateForInstruction i = doInstruction i $ incrementIC u
   35 
   36 ----
   37 
   38 doInstruction :: (SRAutomatonIO Symbol s r m) => Instruction -> Automaton s r -> m $ AutomatonSame s r
   39 doInstruction (IAL      i) u = Left . updateStack   u <$> alInstruction i (unitStack u)
   40 doInstruction (ILS      i) u = Left . updateFromLSU u <$> slInstruction i (toLSU u)
   41 doInstruction (ICF      i) u = Left . updateFromCPU u <$> controlInstruction i (toCPU u)
   42 doInstruction  Transfer    u = transfer u
   43 doInstruction  End         u = end u
   44 
   45 transfer :: (SRAutomatonIO Symbol s r m) => Automaton s r -> m $ AutomatonSame s r
   46 transfer = transferBranch <=< pop2ForStack
   47 
   48 transferBranch :: (SRAutomatonIO Symbol s r m) => (Symbol, Symbol, Automaton s r) -> m $ AutomatonSame s r
   49 transferBranch (_ , 0 , u) = pure $ Left u
   50 transferBranch (0 , _ , u) = end u
   51 transferBranch (a , _ , u) = Left . updateFromCPU u <$> controlInstruction dJumpI (toCPU $ push1ForStack a u)
   52 
   53 pop2ForStack :: (SRAutomatonIO Symbol s r m) => Automaton s r -> m (Symbol , Symbol , Automaton s r)
   54 pop2ForStack u = build <$> pop2 (unitStack u) where
   55   build (s1 , s2 , s') = (s1 , s2 , updateStack u s')
   56 
   57 push1ForStack :: Stack s Symbol => Symbol -> Automaton s r -> Automaton s r
   58 push1ForStack e u = u { unitStack = push1 e (unitStack u) }
   59 
   60 end :: (SRAutomatonIO Symbol s r m) => Automaton s r -> m $ AutomatonSame s r
   61 end = pure . Right

   65 flippedNewAutomaton :: (s , r) -> InstructionList -> Automaton s r
   66 flippedNewAutomaton = flip (uncurry . newAutomaton)
   67 
   68 newAutomaton :: InstructionList -> s -> r -> Automaton s r
   69 newAutomaton il = Automaton (newCU il)
   70 
   71 -- | Updaters
   72 
   73 incrementIC :: Automaton s r -> Automaton s r
   74 incrementIC u = u { unitCU = incrementPC $ unitCU u}
   75 
   76 updateStack :: Automaton s r -> s -> Automaton s r
   77 updateStack u s = u {unitStack = s}
   78 
   79 updateFromCPU :: Automaton s r -> CentralProcessingUnit s -> Automaton s r
   80 updateFromCPU u cpu = u { unitCU = controlUnit cpu, unitStack = alu cpu}
   81 
   82 updateFromLSU :: Automaton s r -> LoadStoreUnit s r -> Automaton s r
   83 updateFromLSU u lsu = u {unitStack = stack lsu , unitRAM = ram lsu}
   84 
   85 -- | Getters
   86 
   87 toCPU :: Automaton s r -> CentralProcessingUnit s
   88 toCPU u = CPU { controlUnit = unitCU u , alu = unitStack u }
   89 
   90 toLSU :: Automaton s r -> LoadStoreUnit s r
   91 toLSU u = LSU { stack = unitStack u, ram = unitRAM u }
   92
