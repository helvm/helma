   16 parse :: MonadSafe m => Source -> m Lambda
   17 parse = parseCode . filterComments
   18 
   19 parseCode :: MonadSafe m => Source -> m Lambda
   20 parseCode = runParser appParser
   21 
   22 appParser :: ReadP Lambda
   23 appParser = foldlLambda <$> manyNonEmpty lambdaParser
   24 
   25 lambdaParser :: ReadP Lambda
   26 lambdaParser =
   27        S <$ oneOf "sS"
   28   <|>  K <$ oneOf "kK"
   29   <|>  I <$ oneOf "iI"
   30   <|>  App <$ char '`' <*> lambdaParser <*> lambdaParser
   31   <|>  char '(' *> appParser <* char ')'
   32 
   33 -- | ParserExtras
   34 runParser :: MonadSafe m => ReadP Lambda -> Source -> m Lambda
   35 runParser parser source = fst . last <$> nonEmptyRunParser parser source
   36 
   37 nonEmptyRunParser :: MonadSafe m => ReadP Lambda -> Source -> m $ NonEmpty (Lambda , String)
   38 nonEmptyRunParser parser source = nonEmptyFromList ("Cannot parse source\n" <> source) $ listRunParser parser source
   39 
   40 listRunParser :: ReadP Lambda -> Source -> [(Lambda , String)]
   41 listRunParser parser = readP_to_S parser . toString
   42 
   43 oneOf :: String -> ReadP Char
   44 oneOf cs = satisfy (`elem` cs)
   45 
   46 manyNonEmpty :: Alternative f => f a -> f $ NonEmpty a
   47 manyNonEmpty p = liftA2 (:|) p (many p)
