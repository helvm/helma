   40 parse :: MonadSafe m => TokenType -> Source -> Bool -> m InstructionList
   41 parse tokenType = flip parseFromTL . tokenize tokenType

   43 parseFromTL :: MonadSafe m => Bool -> TokenList -> m InstructionList
   44 parseFromTL ascii = go where
   45   go :: MonadSafe m => TokenList -> m InstructionList
   46   go      []   = pure              []
   47   go (S:  tl') = stackManipulation tl'
   48   go (T:S:tl') = arithmetic        tl'
   49   go (T:T:tl') = heapAccess        tl'
   50   go (N  :tl') = flowControl       tl'
   51   go (T:N:tl') = ioInstruction     tl'
   52   go      tl'  = notEnoughTokens   tl'
   53 
   54   stackManipulation :: MonadSafe m => TokenList -> m InstructionList
   55   stackManipulation (S:  tl') = go' =<< parseSymbol tl' where go' (symbol , tl'') = (IAL (Cons symbol)         : ) <$> go tl''
   56   stackManipulation (T:S:tl') = go' =<< parseIndex  tl' where go' (index  , tl'') = (IAL (SStatic index Copy)  : ) <$> go tl''
   57   stackManipulation (T:T:tl') = unrecognisedTokens "STT" tl'
   58   stackManipulation (T:N:tl') = go' =<< parseIndex  tl' where go' (index  , tl'') = (IAL (SStatic index Slide) : ) <$> go tl''
   59   stackManipulation (N:S:tl') = (IAL Dup               : ) <$> go tl'
   60   stackManipulation (N:T:tl') = (IAL Swap              : ) <$> go tl'
   61   stackManipulation (N:N:tl') = (IAL Discard           : ) <$> go tl'
   62   stackManipulation      tl'  = notEnoughTokensInGroup "stackManipulation" tl'
   63 
   64   arithmetic :: MonadSafe m => TokenList -> m InstructionList
   65   arithmetic (S:S:tl') = (IAL (Binary Add) : ) <$> go tl'
   66   arithmetic (S:T:tl') = (IAL (Binary Sub) : ) <$> go tl'
   67   arithmetic (S:N:tl') = (IAL (Binary Mul) : ) <$> go tl'
   68   arithmetic (T:S:tl') = (IAL (Binary Div) : ) <$> go tl'
   69   arithmetic (T:T:tl') = (IAL (Binary Mod) : ) <$> go tl'
   70   arithmetic (T:N:tl') = unrecognisedTokens "TSTN" tl'
   71   arithmetic (N:S:tl') = unrecognisedTokens "TSNS" tl'
   72   arithmetic (N:T:tl') = unrecognisedTokens "TSNT" tl'
   73   arithmetic (N:N:tl') = unrecognisedTokens "TSNN" tl'
   74   arithmetic      tl'  = notEnoughTokensInGroup "arithmetic" tl'
   75 
   76   heapAccess :: MonadSafe m => TokenList -> m InstructionList
   77   heapAccess (S:tl') = (ILS Store : ) <$> go tl'
   78   heapAccess (T:tl') = (ILS Load  : ) <$> go tl'
   79   heapAccess (N:tl') = unrecognisedTokens "TTN" tl'
   80   heapAccess    tl'  = notEnoughTokensInGroup "heapAccess" tl'
   81 
   82   flowControl :: MonadSafe m => TokenList -> m InstructionList
   83   flowControl (S:S:tl') = go' =<< parseLabel ascii tl' where go' (label , tl'') = (IControl (Mark    label             ) : ) <$> go tl''
   84   flowControl (S:T:tl') = go' =<< parseLabel ascii tl' where go' (label , tl'') = (IControl (CStatic label Call        ) : ) <$> go tl''
   85   flowControl (S:N:tl') = go' =<< parseLabel ascii tl' where go' (label , tl'') = (IControl (CStatic label Jump        ) : ) <$> go tl''
   86   flowControl (T:S:tl') = go' =<< parseLabel ascii tl' where go' (label , tl'') = (IControl (CStatic label (Branch EZ )) : ) <$> go tl''
   87   flowControl (T:T:tl') = go' =<< parseLabel ascii tl' where go' (label , tl'') = (IControl (CStatic label (Branch LTZ)) : ) <$> go tl''
   88   flowControl (T:N:tl') = (IControl Return : ) <$> go tl'
   89   flowControl (N:S:tl') = unrecognisedTokens "NNS" tl'
   90   flowControl (N:T:tl') = unrecognisedTokens "NNT" tl'
   91   flowControl (N:N:tl') = (End : ) <$> go tl'
   92   flowControl      tl'  = notEnoughTokensInGroup "flowControl" tl'
   93 
   94   ioInstruction :: MonadSafe m => TokenList -> m InstructionList
   95   ioInstruction (S:S:tl') = (IAL (SIO OutputChar) : ) <$> go tl'
   96   ioInstruction (S:T:tl') = (IAL (SIO OutputDec)  : ) <$> go tl'
   97   ioInstruction (S:N:tl') = unrecognisedTokens "TNSN" tl'
   98   ioInstruction (T:S:tl') = (ILS (MIO InputChar)  : ) <$> go tl'
   99   ioInstruction (T:T:tl') = (ILS (MIO InputDec)   : ) <$> go tl'
  100   ioInstruction (T:N:tl') = unrecognisedTokens "TNTN" tl'
  101   ioInstruction (N:S:tl') = unrecognisedTokens "TNNS" tl'
  102   ioInstruction (N:T:tl') = unrecognisedTokens "TNNT" tl'
  103   ioInstruction (N:N:tl') = unrecognisedTokens "TNNN" tl'
  104   ioInstruction tl'       = notEnoughTokensInGroup "ioInstruction" tl'
  105 
  106 unrecognisedTokens :: MonadSafe m => Text -> TokenList -> m InstructionList
  107 unrecognisedTokens tokens tl = liftErrorTupleList [("Unrecognised tokens" , tokens) , ("Rest tokens" , show tl)]
  108 
  109 notEnoughTokensInGroup :: MonadSafe m => Text -> TokenList -> m InstructionList
  110 notEnoughTokensInGroup g tl = liftErrorTupleList [("Not enough tokens in group:" , g) , ("Rest tokens" , show tl)]
  111 
  112 notEnoughTokens :: MonadSafe m => TokenList -> m InstructionList
  113 notEnoughTokens tl = liftErrorTupleList [("Rest tokens" , show tl)]
