   26 parseInt :: MonadSafe m => OperandParser m Int
   27 parseInt tl = parseInt' <$> parseInteger tl where
   28   parseInt' (integer , tl') = (fromIntegral integer , tl')
   29 
   30 parseInteger :: MonadSafe m => OperandParser m Integer
   31 parseInteger []       = liftError "EOL"
   32 parseInteger (S : tl) = parseExtra makeIntegral2FromList tl
   33 parseInteger (T : tl) = negationIntegral <$> parseExtra makeIntegral2FromList tl
   34 parseInteger (N : tl) = pure (0 , tl)
   35 
   36 negationIntegral :: (Integer , TokenList) -> (Integer , TokenList)
   37 negationIntegral (i , l) = (-i , l)
   38 
   39 parseNatural :: MonadSafe m => OperandParser m Natural
   40 parseNatural = parseExtra makeIntegral2FromList
   41 
   42 parseExtra :: MonadSafe m => (TokenList -> m a) -> OperandParser m a
   43 parseExtra maker = loop act . ([] , ) where
   44   act (acc ,      []) = Right $ liftError $ show acc
   45   act (acc ,  N : tl) = Right $ moveSafe (maker acc , tl)
   46   act (acc ,  t : tl) = Left (t : acc , tl)
   47 
   48 parseDigitString :: MonadSafe m => OperandParser m SString
   49 parseDigitString tl = moveSafe =<< parseString' makeDigitStringFromList tl
   50 
   51 parseAsciiString :: MonadSafe m => OperandParser m SString
   52 parseAsciiString tl = moveSafe =<< parseString' makeAsciiString28FromList tl
   53 
   54 moveSafe :: MonadSafe m => (m a , TokenList) -> m (a , TokenList)
   55 moveSafe (a , tl) = appendErrorTuple ("TokenList" , show tl) $ ( , tl) <$> a
   56 
   57 parseString' :: MonadSafe m => (TokenList -> a) -> OperandParser m a
   58 parseString' maker tl = parseString'' <$> splitByN tl where
   59   parseString'' (acc , tl') = (maker acc , tl')
   60 
   61 splitByN :: MonadSafe m => OperandParser m TokenList
   62 splitByN []       = liftError "Empty list"
   63 splitByN (N : tl) = pure ([]    , tl)
   64 splitByN (t : tl) = splitByN' <$> splitByN tl where
   65   splitByN' (acc , tl') = (t:acc , tl')
   66 
   67 -- | Types
   68 type OperandParser m a = TokenList -> m (a , TokenList)
