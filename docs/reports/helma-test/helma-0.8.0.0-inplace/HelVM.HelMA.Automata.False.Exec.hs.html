    9 exec :: BIO m => [Value] -> [Value] -> [Value] -> m ([Value], [Value])
   10 exec [] stack reg = return (stack, reg)
   11 
   12 -- Handle Arithmetic
   13 exec ((Ope Plus    ):prog) ((Num a):(Num b):stack) reg = exec prog ((Num (b+a)):stack) reg
   14 exec ((Ope Minus   ):prog) ((Num a):(Num b):stack) reg = exec prog ((Num (b-a)):stack) reg
   15 exec ((Ope Multiply):prog) ((Num a):(Num b):stack) reg = exec prog ((Num (b*a)):stack) reg
   16 exec ((Ope Division):prog) ((Num a):(Num b):stack) reg = exec prog ((Num (b `div` a)):stack) reg
   17 exec ((Ope Negate  ):prog) ((Num a):stack) reg = exec prog ((Num (negate a)):stack) reg
   18 
   19 -- Handle Comparison
   20 exec ((Ope Equal):prog) ((Num a):(Num b):stack) reg = exec prog ((Num (boolToNum (b == a))):stack) reg
   21 exec ((Ope Greater):prog) ((Num a):(Num b):stack) reg = exec prog ((Num (boolToNum (b > a))):stack) reg
   22 
   23 -- Handle Boolean Algebra
   24 exec ((Ope And):prog) ((Num a):(Num b):stack) reg = exec prog ((Num (b `nAnd` a)):stack) reg
   25 exec ((Ope Or ):prog) ((Num a):(Num b):stack) reg = exec prog ((Num (b `nOr` a)):stack) reg
   26 exec ((Ope Not):prog) ((Num a):stack) reg = exec prog ((Num (nNot a)):stack) reg
   27 
   28 -- Handle Stack Operation
   29 exec ((Ope Dup ):prog) (a : stack) reg = exec prog (a:a:stack) reg
   30 exec ((Ope Drop):prog) (_ : stack) reg = exec prog stack reg
   31 exec ((Ope Swap):prog) (a : b : stack) reg = exec prog (b:a:stack) reg
   32 exec ((Ope Rot ):prog) (a : b : c : stack) reg = exec prog (c:a:b:stack) reg
   33 exec ((Ope Pick):prog) ((Num a) : stack ) reg = exec prog ((pick a stack) : stack) reg
   34 
   35 -- Handle Control Flow
   36 exec ((Ope Execute):prog) ((Fn lambda):stack) reg = do
   37   (newStack, newReg) <- exec lambda stack reg
   38   exec prog newStack newReg
   39 
   40 exec ((Ope If):prog) (fn:(Num a):stack) reg =
   41   if (numToBool a)
   42   then exec ((Ope Execute):prog) (fn:stack) reg
   43   else exec prog stack reg
   44 
   45 exec true@((Ope While):prog) ((Fn body):(Fn cond):stack) reg = do
   46   (condStack, condReg) <- exec cond stack reg
   47   if (while condStack)
   48   then do
   49     (newStack, newReg) <- (exec body stack condReg)
   50     exec true ((Fn body):(Fn cond):newStack) newReg
   51   else exec prog stack condReg
   52 
   53 -- Handle Standard Input/Output
   54 exec ((Io ReadChar):prog) stack reg = do
   55   char <- wGetChar
   56   exec prog ((Num (charToNum char)):stack) reg
   57 exec ((Io WriteChar):prog) ((Num a):stack) reg = do
   58   wPutChar $ numToChar a
   59   exec prog stack reg
   60 exec ((Io WriteDec):prog) ((Num a):stack) reg = do
   61   wPutStr $ show a
   62   exec prog stack reg
   63 exec ((Io (Str str)):prog) stack reg = do
   64   wPutStr $ toText str
   65   exec prog stack reg
   66 
   67 -- Handle Variables Operations
   68 exec ((Ope Store):prog) ((Var key):val:stack) reg = exec prog stack (putByIndex reg key val)
   69 exec ((Ope Fetch):prog) ((Var key):stack) reg = exec prog ((getByIndex reg key):stack) reg
   70 
   71 -- Handle Values
   72 exec ((Ope op):prog) stack reg = error ("Invalid operator parameters: " <> (show op) <> ", nprog " <> (show prog) <> ", stack " <> (show stack) <> ", reg" <> (show reg)) --(exec prog stack reg)
   73 --exec ((Ope _):_) stack reg = pure $ (stack , reg)
   74 exec ((Ch c):prog)   stack reg = exec prog ((Num (charToNum c)):stack) reg
   75 exec (a:prog)        stack reg = exec prog (a:stack) reg
   76 
   77 -- FALSE operators
   78 pick :: Integer -> [Value] -> Value
   79 pick n stack = Unsafe.head $ drop (fromInteger n) stack
   80 
   81 while :: [Value] -> Bool
   82 while = valueToBool . Unsafe.head
   83 
   84 putByIndex' :: [Value] -> [Value] -> Value -> Int -> [Value]
   85 putByIndex' left (  _ : right) val   0 = left <> [val] <> right
   86 putByIndex' left (mov : right) val key = putByIndex' (left<>[mov]) right val (key-1)
   87 putByIndex'    _            []   _   _ = error "Empty"
   88 
   89 
   90 putByIndex :: [Value] -> Int -> Value -> [Value]
   91 putByIndex reg key val = putByIndex' [] reg val key
   92 
   93 getByIndex :: [Value] -> Int -> Value
   94 getByIndex (val :   _)   0 = val
   95 getByIndex (  _ : reg) key = getByIndex reg (key - 1)
   96 getByIndex []            _ = error "Empty"
   97 
   98 -- Boolean Num Operators
   99 nAnd :: Integer -> Integer -> Integer
  100 nAnd 0 0     = 0
  101 nAnd   _ 0   = 0
  102 nAnd 0   _   = 0
  103 nAnd   _   _ = 1
  104 
  105 nNot :: Integer -> Integer
  106 nNot 0   = 1
  107 nNot   _ = 0
  108 
  109 nOr :: Integer -> Integer -> Integer
  110 nOr 0 0     = 0
  111 nOr   _ 0   = 1
  112 nOr 0   _   = 1
  113 nOr   _   _ = 1
