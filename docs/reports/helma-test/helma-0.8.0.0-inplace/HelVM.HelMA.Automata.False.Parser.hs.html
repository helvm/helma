   14 parse :: [Char] -> [Value]
   15 parse [] = []
   16 -- Parse Arithmetic
   17 parse ('+':prog) = (Ope Plus):(parse prog)
   18 parse ('-':prog) = (Ope Minus):(parse prog)
   19 parse ('*':prog) = (Ope Multiply):(parse prog)
   20 parse ('/':prog) = (Ope Division):(parse prog)
   21 parse ('_':prog) = (Ope Negate):(parse prog)
   22 
   23 -- Parse Comparison
   24 parse ('=':prog) = (Ope Equal):(parse prog)
   25 parse ('>':prog) = (Ope Greater):(parse prog)
   26 
   27 -- Parse Boolean Algebra
   28 parse ('&':prog) = (Ope And):(parse prog)
   29 parse ('|':prog) = (Ope Or):(parse prog)
   30 parse ('~':prog) = (Ope Not):(parse prog)
   31 
   32 -- Parse Stack Ope
   33 parse ('$':prog) = (Ope Dup):(parse prog)
   34 parse ('%':prog) = (Ope Drop):(parse prog)
   35 parse ('\\':prog) = (Ope Swap):(parse prog)
   36 parse ('@':prog) = (Ope Rot):(parse prog)
   37 parse ('o':prog) = (Ope Pick):(parse prog)
   38 
   39 -- Parse Control Flow
   40 parse ('!':prog) = (Ope Execute):(parse prog)
   41 parse ('?':prog) = (Ope If):(parse prog)
   42 parse ('#':prog) = (Ope While):(parse prog)
   43 parse ('{':prog) = (parse (afterComment prog))
   44 parse ('[':prog) = (Fn (parse $ parseLambda 0 prog)):(parse $ afterLambda 0 prog)
   45 
   46 -- Parse Standard Input/Output
   47 parse ('^':prog) = (Io ReadChar):(parse prog)
   48 parse (',':prog) = (Io WriteChar):(parse prog)
   49 parse ('.':prog) = (Io WriteDec):(parse prog)
   50 parse ('"':prog) = (Io (Str (parseString prog))):(parse $ afterString prog)
   51 
   52 -- Parse Variables
   53 parse (c:prog) | isAlpha c = (Var (charToVar c)):(parse prog)
   54 parse (':':prog) = (Ope Store):(parse prog)
   55 parse (';':prog) = (Ope Fetch):(parse prog)
   56 
   57 -- Parse Literals
   58 parse ('\'':c:prog) = (Ch c):(parse prog)
   59 parse full@(i:prog) | isDigit i = (Num (readTextUnsafe $ toText $ parseNum full)):(parse (afterNum prog))
   60 
   61 -- Skip other shit
   62 parse (_:prog) = (parse prog)

   65 -- Parse manipulators
   66 -- Parse Strings
   67 parseString :: [Char] -> [Char]
   68 parseString ('"':_)         = []
   69 -- Handle special characters explicitly
   70 parseString ('\\':'a':prog) = (chr 7):(parseString prog)
   71 parseString ('\\':'b':prog) = (chr 8):(parseString prog)
   72 parseString ('\\':'f':prog) = (chr 12):(parseString prog)
   73 parseString ('\\':'n':prog) = (chr 10):(parseString prog)
   74 parseString ('\\':'r':prog) = (chr 13):(parseString prog)
   75 parseString ('\\':'t':prog) = (chr 9):(parseString prog)
   76 parseString ('\\':'"':prog) = (chr 34):(parseString prog)
   77 parseString (c:prog)        = c:(parseString prog)
   78 parseString []              = error "Empty"
   79 
   80 afterString :: [Char] -> [Char]
   81 afterString ('"' : prog) = prog
   82 afterString (_   : prog) = afterString prog
   83 afterString []           = error "Empty"
   84 
   85 
   86 -- Parse Lambdas
   87 parseLambda :: Int -> [Char] -> [Char]
   88 parseLambda _ []         = error "Unclosed lambda"
   89 parseLambda 0 (']':_)    = []
   90 parseLambda n (']':prog) = ']':(parseLambda (n-1) prog)
   91 parseLambda n ('[':prog) = '[':(parseLambda (n+1) prog)
   92 parseLambda n (c:prog)   = c:(parseLambda n prog)
   93 
   94 afterLambda :: Int -> [Char] -> [Char]
   95 afterLambda _ []         = error "Unclosed lambda"
   96 afterLambda 0 (']':prog) = prog
   97 afterLambda n (']':prog) = afterLambda (n-1) prog
   98 afterLambda n ('[':prog) = afterLambda (n+1) prog
   99 afterLambda n (_:prog)   = afterLambda n prog
  100 
  101 -- Parse Numbers
  102 parseNum :: [Char] -> [Char]
  103 parseNum [] = []
  104 parseNum ('.':i:prog) | isDigit i = '.':i:(parseNum prog)
  105 parseNum (i:prog) | isDigit i = i:(parseNum prog)
  106 parseNum (_ : _) = []
  107 
  108 afterNum :: [Char] -> [Char]
  109 afterNum [] = []
  110 afterNum (i:prog) | isDigit i = afterNum prog
  111 afterNum full@(_:_) = full
  112 
  113 
  114 -- Ignore Comments
  115 afterComment :: [Char] -> [Char]
  116 afterComment []         = error "Unclosed Comment"
  117 afterComment ('}':prog) = prog
  118 afterComment (_:prog)   = afterComment prog
