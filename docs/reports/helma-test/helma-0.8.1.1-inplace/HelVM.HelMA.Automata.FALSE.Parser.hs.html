   21 parseSafe :: Source -> Safe ValueList
   22 parseSafe = parse
   23 
   24 parse :: MonadSafe m => Source -> m ValueList
   25 parse = runParser vlParser
   26 
   27 vlParser :: ReadP ValueList
   28 vlParser = many (skipSpaces *> valueParser) <* skipSpaces
   29 
   30 valueParser :: ReadP Value
   31 valueParser =
   32       constParser
   33   <|> dupParser <|> dropParser <|> swapParser <|> rotParser <|> pickParser
   34   <|> addParser <|> subParser <|> mulParser <|> divParser <|> negParser
   35   <|> andParser <|> orParser <|> notParser
   36   <|> gtParser <|> eqParser
   37   <|> lambdaParser <|> execParser <|> condParser <|> whileParser
   38   <|> refParser <|> stParser <|> ldParser
   39   <|> readCharParser <|> writeCharParser <|> writeStringParser <|> writeNumParser <|> flushParser
   40   <|> commentParser
   41 
   42 constParser :: ReadP Value
   43 constParser = Inst . IAL . Cons . fromIntegral <$> naturalParser
   44 
   45 dupParser , dropParser , swapParser , rotParser , pickParser :: ReadP Value
   46 dupParser  = Inst (IAL dupI           ) <$ char '$'
   47 dropParser = Inst (IAL Discard        ) <$ char '%'
   48 swapParser = Inst (IAL swapI          ) <$ char '\\'
   49 rotParser  = Inst (IAL rotI           ) <$ char '@'
   50 pickParser = Inst (IAL (SDynamic Copy)) <$ char '`'
   51 
   52 addParser , subParser , mulParser , divParser , negParser :: ReadP Value
   53 addParser = Inst (IAL (Binary Add)) <$ char '+'
   54 subParser = Inst (IAL (Binary Sub)) <$ char '-'
   55 mulParser = Inst (IAL (Binary Mul)) <$ char '*'
   56 divParser = Inst (IAL (Binary Div)) <$ char '/'
   57 negParser = Inst (IAL (Unary  Neg)) <$ char '_'
   58 
   59 andParser , orParser , notParser :: ReadP Value
   60 andParser = Inst (IAL (Binary BAnd)) <$ char '&'
   61 orParser  = Inst (IAL (Binary BOr )) <$ char '|'
   62 notParser = Inst (IAL (Unary  BNot)) <$ char '~'
   63 
   64 gtParser , eqParser :: ReadP Value
   65 gtParser = Inst (IAL (Binary LGT)) <$ char '<'
   66 eqParser = Inst (IAL (Binary LEQ)) <$ char '='
   67 
   68 lambdaParser , execParser , condParser , whileParser  :: ReadP Value
   69 lambdaParser = Lambda <$> (char '[' *> vlParser <* char ']')
   70 execParser   = Exec  <$ char '!'
   71 condParser   = Cond  <$ char '?'
   72 whileParser  = While <$ char '#'
   73 
   74 refParser , stParser , ldParser :: ReadP Value
   75 refParser = refFromChar <$> letterAscii
   76 stParser  = Store <$ char ':'
   77 ldParser  = Fetch <$ char ';'
   78 
   79 readCharParser , writeCharParser , writeNumParser , writeStringParser , flushParser:: ReadP Value
   80 readCharParser    = Inst (IAL (SIO InputChar )) <$ char '^'
   81 writeCharParser   = Inst (IAL (SIO OutputChar)) <$ char ','
   82 writeStringParser = Str <$> stringParser
   83 writeNumParser    = Inst (IAL (SIO OutputDec )) <$ char '.'
   84 flushParser       = Flush <$ char 'ÃŸ'
   85 
   86 commentParser :: ReadP Value
   87 commentParser = Comment <$> (char '{' *> many (notChar '}') <* char '}')
   88 
   89 -- | Extra
   90 
   91 refFromChar :: Char -> Value
   92 refFromChar c = Ref $ fromIntegral $ ord (toLower c) - ord 'a'
   93 
   94 naturalParser :: ReadP Natural
   95 naturalParser = naturalLiteralParser <|> ordCharLiteralParser
   96 
   97 naturalLiteralParser :: ReadP Natural
   98 naturalLiteralParser = readUnsafe <$> many1 digit
   99 
  100 ordCharLiteralParser :: Integral a => ReadP a
  101 ordCharLiteralParser = fromIntegral . ord <$> (skipSpacesAndChar '\'' *> anyChar)
  102 
  103 stringParser :: ReadP String
  104 stringParser = skipSpacesAndChar '"' *> many (notChar '"') <* char '"'
  105 
  106 skipSpacesAndChar :: Char -> ReadP Char
  107 skipSpacesAndChar c = skipSpaces *> char c
