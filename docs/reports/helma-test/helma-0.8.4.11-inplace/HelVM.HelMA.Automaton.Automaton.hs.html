   41 start :: BIO m => InstructionList -> AutomatonOptions -> m ()
   42 start il ao = start' (flip optimize il $ optLevelAutoOptions ao) (stackType ao) (ramType ao) (autoOptions ao)
   43 
   44 start' :: BIO m => InstructionList -> StackType -> RAMType -> AutoOptions -> m ()
   45 start' il s ListRAMType    = start'' il s []
   46 start' il s SeqRAMType     = start'' il s Seq.empty
   47 start' il s SListRAMType   = start'' il s SList.sListEmpty
   48 start' il s MapListRAMType = start'' il s MapList.mapListEmpty
   49 
   50 start'' :: (RAutomatonIO Symbol r m) => InstructionList -> StackType -> r -> AutoOptions -> m ()
   51 start'' il ListStackType  = start''' il []
   52 start'' il SeqStackType   = start''' il Seq.empty
   53 start'' il SListStackType = start''' il SList.sListEmpty
   54 
   55 start''' :: (SRAutomatonIO Symbol s r m) => InstructionList -> s -> r -> AutoOptions -> m ()
   56 start''' il s r p = runAndDumpLogs p (newMemory il s r)
   57 
   58 runAndDumpLogs :: (SRAutomatonIO Symbol s r m) => AutoOptions -> Memory s r ->  m ()
   59 runAndDumpLogs p m = logDump (dumpType p) =<< runWithLimit ((limit p) , m)
   60 
   61 run :: (SRAutomatonIO Symbol s r m) => LimitMaybe -> F s r m
   62 --run = trampolineMWithLimit nextState
   63 run l m = snd <$> runWithLimit (l , m)
   64 
   65 runWithLimit :: (SRAutomatonIO Symbol s r m) => (WithLimitMaybe (Memory s r)) -> m $ WithLimitMaybe (Memory s r)
   66 runWithLimit = trampolineMWithLimit2 nextState
   67 
   68 nextState :: (SRAutomatonIO Symbol s r m) => SF s r m
   69 nextState a = nextStateForInstruction =<< currentInstruction (memoryCM a) where
   70   nextStateForInstruction i = appendErrorTuple ("Automaton.nextState" , showP a) $ appendErrorTuple ("program:" , printIndexedIL $ toList program) $ appendErrorTuple ("i:" , show i) $ runInstruction i $ incrementIC a where
   71     program = memoryProgram a
