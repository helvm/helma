   22 next :: (SEvaluator e s m) => Unit s -> m $ Unit s
   23 next = loopM act
   24 
   25 act :: (SEvaluator e s m) => Unit s -> m $ UnitBoth s
   26 act (Unit iu s) = build =<< nextIU iu  where build (t , iu') = doInstruction t (Unit iu' s)
   27 
   28 doInstruction :: (SEvaluator e s m) => Maybe Token -> Unit s -> m $ UnitBoth s
   29 -- | IO instructions
   30 doInstruction (Just O) u                      = Left . updateStack u <$> doOutputChar2 (unitStack u)
   31 doInstruction (Just I) u                      = Left . updateStack u <$> doInputChar2 (unitStack u)
   32 
   33 -- | Stack instructions
   34 doInstruction (Just N) (Unit iu s)            = build <$> parseNumber iu where build (symbol , iu') = Left (Unit iu' (push1 symbol s))
   35 doInstruction (Just H) u                      = Left . updateStack u <$> halibut (unitStack u)
   36 
   37 -- | Arithmetic
   38 doInstruction (Just S) u                      = Left . updateStack u <$> sub (unitStack u)
   39 doInstruction (Just E) u                      = Left . updateStack u <$> divMod (unitStack u)
   40 
   41 -- | Control
   42 doInstruction (Just R) u                      = pure $ Left u
   43 doInstruction (Just A) (Unit iu@(IU il ic) s) = pure $ Left ((Unit iu . flipPush1 s . genericNextLabel il) ic)
   44 doInstruction (Just T) u                      = transfer u
   45 doInstruction Nothing u                       = end u
   46 
   47 transfer :: (SEvaluator e s m) => Unit s -> m $ UnitBoth s
   48 transfer = branch <=< pop2ForStack where
   49   branch (_ , 0 , u) = pure $ Left u
   50   branch (0 , _ , u) = end u
   51   branch (l , _ , u) = Left . updateAddress u <$> genericFindAddress (unitProgram u) l
   52 
   53 pop2ForStack :: (SEvaluator e s m) => Unit s -> m (e , e , Unit s)
   54 pop2ForStack u = build <$> pop2 (unitStack u) where
   55   build (s1 , s2 , s') = (s1 , s2 , updateStack u s')
   56 
   57 -- | Terminate instruction
   58 end :: (SEvaluator e s m) => Unit s -> m $ UnitBoth s
   59 end = pure . Right

   63 newUnit :: TokenList -> s -> Unit s

   66 updateStack :: Unit s -> s -> Unit s
   67 updateStack u s =  u {unitStack = s}
   68 
   69 updateAddress :: Unit s -> InstructionCounter -> Unit s
   70 updateAddress u a =  u {unitIU = updatePC (unitIU u) a}

   72 unitProgram :: Unit s -> TokenVector
   73 unitProgram = program . unitIU
