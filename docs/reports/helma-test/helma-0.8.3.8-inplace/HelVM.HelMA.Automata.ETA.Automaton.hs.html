   42 run :: (Evaluator Symbol m) => ETAImplType -> Source -> StackType -> Maybe Natural -> DumpType -> m ()
   43 run etaImplType source = evalTL etaImplType (tokenize source)
   44 
   45 evalTL :: (Evaluator Symbol m) => ETAImplType -> TokenList -> StackType -> Maybe Natural -> DumpType -> m ()
   46 evalTL c tl ListStackType  = start c tl []
   47 evalTL c tl SeqStackType   = start c tl Seq.empty
   48 evalTL c tl SListStackType = start c tl SList.sListEmpty
   49 
   50 start :: (SEvaluator Symbol s m) => ETAImplType -> TokenList -> s -> Maybe Natural -> DumpType -> m ()
   51 start Original tl s _     dt = logDump dt =<< next (IU (Vector.fromList tl) 0) s
   52 start Fast     tl s limit dt = Automaton.startWithIL s [] limit dt =<< optimize tl
