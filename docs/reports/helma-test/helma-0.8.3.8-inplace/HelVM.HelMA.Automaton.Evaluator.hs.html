   24 eval :: (SRAutomaton Symbol s r m) => Maybe Natural -> Unit s r -> m $ Unit s r
   25 eval Nothing  = evalWithoutLimit
   26 eval (Just n) = evalWithLimit n
   27 
   28 evalWithLimit :: (SRAutomaton Symbol s r m) => Natural -> Unit s r -> m $ Unit s r
   29 evalWithLimit n u = loopM (actMWithLimit nextUnit) (n , u)
   30 
   31 actMWithLimit :: Monad m => (a -> m $ Both a) -> WithLimit a -> m (Either (WithLimit a) a)
   32 actMWithLimit f (n , x) = checkN n where
   33   checkN 0 = pure $ Right x
   34   checkN _ = mapLeft (n - 1 , ) <$> f x
   35 
   36 evalWithoutLimit ::  (SRAutomaton Symbol s r m) => Unit s r -> m $ Unit s r
   37 evalWithoutLimit = loopM nextUnit

   41 nextUnit :: (SRAutomaton Symbol s r m) => Unit s r -> m $ UnitBoth s r
   42 nextUnit u@(Unit cu _ _) = nextUnitForInstruction =<< currentInstruction cu where
   43   nextUnitForInstruction i = doInstruction i $ incrementIC u
   44

   47 doInstruction :: (SRAutomaton Symbol s r m) => Instruction -> Unit s r -> m $ UnitBoth s r
   48 doInstruction (IAL      i) u = Left . updateStack   u <$> alInstruction i (unitStack u)
   49 doInstruction (ILS      i) u = Left . updateFromLSU u <$> slInstruction i (toLSU u)
   50 doInstruction (ICF      i) u = Left . updateFromCPU u <$> controlInstruction i (toCPU u)
   51 doInstruction  Transfer    u =  transfer u
   52 doInstruction  End         u =  end u
   53 
   54 transfer :: (SRAutomaton Symbol s r m) => Unit s r -> m $ UnitBoth s r
   55 transfer u = branch =<< pop2ForStack u where
   56   branch (_ , 0 , u') = pure $ Left u'
   57   branch (0 , _ , u') = end u'
   58   branch (a , _ , u') = Left . updateFromCPU u' <$> controlInstruction dJumpI (toCPU $ push1ForStack a u')
   59 
   60 pop2ForStack :: (SRAutomaton Symbol s r m) => Unit s r -> m (Symbol , Symbol , Unit s r)
   61 pop2ForStack u = build <$> pop2 (unitStack u) where
   62   build (s1 , s2 , s') = (s1 , s2 , updateStack u s')
   63 
   64 push1ForStack :: Stack s Symbol => Symbol -> Unit s r -> Unit s r
   65 push1ForStack e u = u { unitStack = push1 e (unitStack u) }
   66 
   67 end :: (SRAutomaton Symbol s r m) => Unit s r -> m $ UnitBoth s r
   68 end = pure . Right
