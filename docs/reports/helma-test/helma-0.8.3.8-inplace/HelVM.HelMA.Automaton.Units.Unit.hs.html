    8 -- | Constructors
    9 
   10 newUnit :: InstructionList -> s -> r -> Unit s r
   11 newUnit il = Unit (newCU il)
   12 
   13 -- | Updaters
   14 
   15 incrementIC :: Unit s r -> Unit s r
   16 incrementIC u = u { unitCU = incrementPC $ unitCU u}
   17 
   18 updateStack :: Unit s r -> s -> Unit s r
   19 updateStack u s = u {unitStack = s}
   20 
   21 updateFromCPU :: Unit s r -> CentralProcessingUnit s -> Unit s r
   22 updateFromCPU u cpu = u { unitCU = controlUnit cpu, unitStack = alu cpu}
   23 
   24 updateFromLSU :: Unit s r -> LoadStoreUnit s r -> Unit s r
   25 updateFromLSU u lsu = u {unitStack = stack lsu , unitRAM = ram lsu}
   26 
   27 -- | Getters
   28 
   29 toCPU :: Unit s r -> CentralProcessingUnit s
   30 toCPU u = CPU { controlUnit = unitCU u , alu = unitStack u }
   31 
   32 toLSU :: Unit s r -> LoadStoreUnit s r
   33 toLSU u = LSU { stack = unitStack u, ram = unitRAM u }
   34 
   35 -- | Data types
   36 data Unit s r = Unit
   37   { unitCU    :: ControlUnit
   38   , unitStack :: s
   39   , unitRAM   :: r
   40   }
