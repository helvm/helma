   48 moveHead = changeTape moveHeadRight moveHeadLeft

   50 changeTape :: FullTapeD e -> FullTapeD e -> Int -> FullTapeD e
   51 changeTape lf gf i t = loop atc (i , t) where
   52   atc (i' , t') = (check . compare0) i' where
   53     check LT = Left (i' - 1 , lf t')
   54     check GT = Left (i' + 1 , gf t')
   55     check EQ = Right t'
   56 
   57 moveHeadRight :: (Symbol e) => FullTapeD e
   58 moveHeadRight (cell : left , right) = pad (left , cell : right)
   59 moveHeadRight ([] , _)              = error "End of the Tape"
   60 
   61 moveHeadLeft :: (Symbol e) => FullTapeD e
   62 moveHeadLeft (left , cell : right) = pad (cell : left , right)
   63 moveHeadLeft (_ , [])              = error "End of the Tape"

   65 pad :: (Symbol e) => FullTapeD e
   66 pad ([] , [])    = newTape
   67 pad ([] , right) = ([def] , right)
   68 pad (left , [])  = (left , [def])
   69 pad tape         = tape
