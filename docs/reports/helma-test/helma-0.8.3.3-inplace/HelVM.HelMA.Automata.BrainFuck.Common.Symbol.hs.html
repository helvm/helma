   19 inc :: Symbol e => Int -> e -> e
   20 inc i e = loop atc (i , e) where
   21   atc (i' , e') = (check . compare0) i' where
   22     check LT = Left (i' - 1 , next e')
   23     check GT = Left (i' + 1 , prev e')
   24     check EQ = Right e'
