   48 runWithParams :: BIO m => TokenType -> EvalParams -> m ()
   49 runWithParams tokenType p = run tokenType (source p) (formatType p) (stackTypeOptions p) (ramTypeOptions p) Nothing (dumpTypeOptions p)

   51 run :: BIO m => TokenType -> Source -> FormatType -> StackType -> RAMType -> Maybe Natural -> DumpType -> m ()
   52 run tokenType source = runTL $ tokenize tokenType source
   53 
   54 runTL :: BIO m => TokenList -> FormatType -> StackType -> RAMType -> Maybe Natural -> DumpType -> m ()
   55 runTL tl ascii st rt limit dt = runTL' =<< liftSafe (parseFromTL ascii tl) where runTL' il = runIL il st rt limit dt
   56 
   57 runIL :: BIO m => InstructionList -> StackType -> RAMType -> Maybe Natural -> DumpType -> m ()
   58 runIL il s ListRAMType    = runIL' il s []
   59 runIL il s SeqRAMType     = runIL' il s Seq.empty
   60 runIL il s SListRAMType   = runIL' il s SList.sListEmpty
   61 runIL il s MapListRAMType = runIL' il s MapList.mapListEmpty
   62 
   63 runIL' :: (REvaluator Symbol r m) => InstructionList -> StackType -> r -> Maybe Natural -> DumpType -> m ()
   64 runIL' il ListStackType  = start il []
   65 runIL' il SeqStackType   = start il Seq.empty
   66 runIL' il SListStackType = start il SList.sListEmpty
